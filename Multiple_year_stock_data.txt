'The ticker symbol'Yearly change from the opening price at the beginning of a given year to the closing price at the end of that year.'The percentage change from the opening price at the beginning of a given year to the closing price at the end of that year.'The total stock volume of the stock.Sub stockAnalyzer():    For Each ws In Worksheets            'Initialize variables for math and outputs        Dim tickerSymbol As String        Dim openPrice As Double        Dim closePrice As Double        Dim highPrice As Double        Dim lowPrice As Double        Dim volume As Double            'Intialize looping variable        Dim i As Long        Dim j As Long            'Assign Variables        openPrice = ws.Range("C2").Value        tickerSymbol = ws.Range("A2").Value        highPrice = ws.Range("D2").Value        lowPrice = ws.Range("E2").Value        volume = 0            'Initialize and assign ticker number to show which variable is on        Dim tick As Integer        tick = 2            'Initialize variable for end of loop (last row number)        Dim last As Long            'Hard Code Header        ws.Range("I1").Value = "Ticker"        ws.Range("J1").Value = "Yearly Change"        ws.Range("K1").Value = "Percent Change"        ws.Range("L1").Value = "Total Stock Volume"            'Find last row and assign to last variable        last = ws.Range("A" & Rows.Count).End(xlUp).Row        'Because of overflow error, made last reasonable number        'last = 10000            'loop thru all rows in sheet        For i = 2 To last                    'If same stock as above stock            If ws.Range("A" & i).Value = tickerSymbol Then                'reassignClose Value                closePrice = ws.Range("F" & i).Value                            'increase total volume of stock purchased                volume = volume + ws.Range("G" & i).Value                            'Change high price if new price is higher                If highPrice < ws.Range("D" & i).Value Then                    highPrice = ws.Range("D" & i).Value                End If                            'Change low price if new low price is lower                If lowPrice > ws.Range("E" & i).Value Then                    lowPrice = ws.Range("E" & i).Value                End If                        Else                'Display Ticker Symbol                ws.Range("I" & tick).Value = tickerSymbol                        'Display Yearly Change                ws.Range("J" & tick).Value = closePrice - openPrice                            'Conditional Formatting                If (closePrice - openPrice) < 0 Then                    ws.Range("J" & tick).Interior.ColorIndex = 3                Else                    ws.Range("J" & tick).Interior.ColorIndex = 4                End If                            'Display Percent Change                ws.Range("K" & tick).Value = (closePrice - openPrice) / openPrice                            'Percent Formatting                ws.Range("K" & tick).NumberFormat = "0.00%"                        'Display Total Stock Volume                ws.Range("L" & tick).Value = volume                            'Assign Given Variables from current row                openPrice = ws.Range("C" & i).Value                tickerSymbol = ws.Range("A" & i).Value                highPrice = ws.Range("D" & i).Value                lowPrice = ws.Range("E" & i).Value                volume = ws.Range("G" & i).Value                tick = tick + 1            End If                       Next i                'Greatest % increase, Greatest % Decrease and Greatest total Volume            'Intialize variables        Dim greatestIncrease As Double        Dim greatestIncreaseSTR As String        Dim greatestDecrease As Double        Dim greatestDecreaseSTR As String        Dim greatestTV As Double        Dim greatestTVSTR As String            'Assign variables        greatestIncrease = 0        greatestDecrease = 0        greatestTV = 0            'find last ticker again but for new data        last = ws.Range("I" & Rows.Count).End(xlUp).Row            'New for loop going thu new data        For j = 2 To last            'check if current row percentage is biggest            If ws.Range("K" & j).Value > greatestIncrease Then                greatestIncrease = ws.Range("K" & j).Value                greatestIncreaseSTR = ws.Range("I" & j).Value            End If                    'check if current row percentage is smallest            If ws.Range("K" & j).Value < greatestDecrease Then                greatestDecrease = ws.Range("K" & j).Value                greatestDecreaseSTR = ws.Range("I" & j).Value            End If                    'check if current row total volume is biggest            If ws.Range("L" & j).Value > greatestTV Then                greatestTV = ws.Range("L" & j).Value                greatestTVSTR = ws.Range("I" & j).Value            End If                Next j            'Hardcode Titles for Greatest values        ws.Range("O2").Value = "Greatest % Increase"        ws.Range("O3").Value = "Greatest % Decrease"        ws.Range("O4").Value = "Greatest Total Volume"        ws.Range("P1").Value = "Ticker"        ws.Range("Q1").Value = "Value"            'Assign Greatest values to be visible in exceldoc        ws.Range("P2").Value = greatestIncreaseSTR        ws.Range("Q2").Value = greatestIncrease        ws.Range("Q2").NumberFormat = "0.00%"        ws.Range("P3").Value = greatestDecreaseSTR        ws.Range("Q3").Value = greatestDecrease        ws.Range("Q3").NumberFormat = "0.00%"        ws.Range("P4").Value = greatestTVSTR        ws.Range("Q4").Value = greatestTV        'move on to next sheet    Next ws    End Sub
